    ```javascript
    function teamDetailApp() {
        return {
            teamId: '{{ team_id }}',
            team: { name: '', execution_backend: 'paper', asset_class: '', watchlist: [], risk_limits: {} },
            portfolio: { total_value: 10000, cash: 10000, unrealized_pnl: 0, total_return_pct: 0, positions: [] },
            sessions: [],
            async init() {
                const [tRes, pRes, sRes] = await Promise.all([
                    fetch('/api/teams/' + this.teamId),
                    fetch('/api/portfolio?team_id=' + this.teamId),
                    fetch('/api/sessions?team_id=' + this.teamId),
                ]);
                if (tRes.ok) this.team = await tRes.json();
                if (pRes.ok) this.portfolio = await pRes.json();
                if (sRes.ok) this.sessions = await sRes.json();
            },
            fmt(n) { return (n || 0).toLocaleString(undefined, { minimumFractionDigits: 2, maximumFractionDigits: 2 }); },
            fmtPct(n) { return (n >= 0 ? '+' : '') + (n || 0).toFixed(2) + '%'; },
        };
    }
    ```

    ```javascript
    function summaryApp() {
        return {
            aggregate: { total_value: 0, cash: 0, unrealized_pnl: 0, realized_pnl: 0, total_return_pct: 0 },
            teams: [],
            async init() {
                const res = await fetch('/api/teams/summary');
                if (res.ok) {
                    const data = await res.json();
                    this.aggregate = data.aggregate;
                    this.teams = data.teams;
                }
            },
            fmt(n) { return (n || 0).toLocaleString(undefined, { minimumFractionDigits: 2, maximumFractionDigits: 2 }); },
            fmtPct(n) { return (n >= 0 ? '+' : '') + (n || 0).toFixed(2) + '%'; },
        };
    }
    ```
